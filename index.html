{

 else if(M>H){

Q3. Perfect Number Check.


     
var Perfect_Check = (N) => 
{
  
  let number=0;

    for(let i=0;i < N ;i++){
      if(N%i===0){
        number+=i;
      }
    }  
    if(number==N){
      return "YES";
    }
    else{
      return "NO";
    }
        
};
